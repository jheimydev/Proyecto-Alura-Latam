import matplotlib.pyplot as plt

# Concatenar todos los datos
df = pd.concat([tienda, tienda2, tienda3, tienda4])
df.reset_index(drop=True, inplace=True)

1. Anlisis de facturacin

facturacion = [
    tienda['Precio'].sum(),
    tienda2['Precio'].sum(),
    tienda3['Precio'].sum(),
    tienda4['Precio'].sum()
]

plt.bar(['Tienda 1', 'Tienda 2', 'Tienda 3', 'Tienda 4'], facturacion, color='green')
plt.title('Facturacin total por tienda')
plt.ylabel('Total en Ventas (Precio)')
plt.show()

2. Ventas por categora

ventas_categoria = df['Categora del Producto'].value_counts()

ventas_categoria.plot(kind='bar', color='orange')
plt.title('Ventas por categora')
plt.ylabel('Cantidad de ventas')
plt.xlabel('Categora')
plt.show()

3. Calificacin promedio de la tienda

calificacion_prom = [
    tienda['Calificacin'].mean(),
    tienda2['Calificacin'].mean(),
    tienda3['Calificacin'].mean(),
    tienda4['Calificacin'].mean()
]

plt.bar(['Tienda 1', 'Tienda 2', 'Tienda 3', 'Tienda 4'], calificacion_prom, color='purple')
plt.title('Calificacin promedio por tienda')
plt.ylabel('Calificacin (0 a 5)')
plt.ylim(0, 5)
plt.show()

4. Productos ms y menos vendidos
